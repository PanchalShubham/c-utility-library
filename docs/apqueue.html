utility/apqueue.h

[since release@2020.2]

Resizable-array implementation of priority queue using binary heaps!
ArrayPriorityQueue is not synchronized!
Capacity of the queue is doubled if array priority queue is 100% filled!
Capacity of the queue is halved if array priority queue is at most 25% filled!
Theoretical maximum capacity of array priority queue is INT_MAX (2,147,483,647 on most platforms)
All memory allocation operations are done on heap!
Insertion and deletion takes logarithmic time!
Searching takes linear time!
since release@2020.2

Function summary

/*creates a new array priority queue for on heap and returns a pointer to it if succeeds otherwise returns NULL*/
/*asserts that initial capacity is a positive integer and type is valid*/
/*since release@2020.2*/
ArrayPriorityQueue* new_apq(int initial_capacity, enum ArrayPriorityQueueType type);

/*adds given data to array priority queue and returns true if succeeds otherwise returns false*/
/*asserts that queue, queue's array and callback aren't NULL*/
/*since release@2020.2*/
bool apq_add(ArrayPriorityQueue* queue, void* data, Comparator comparator);

/*returns and removes the data with max/min priority in priority queue if exist otherwise returns NULL*/
/*asserts that queue, queue's array and callback aren't NULL*/
/*since release@2020.2*/
void* apq_extract(ArrayPriorityQueue* queue, Comparator comparator);

/*replace the 1st occurrence of key with newKey in priority if exist; returns true if succeeds otherwise returns false*/
/*optionally frees the memory allocated to key and keyInQueue*/
/*asserts that queue, queue's array, callback aren't NULL*/
/*since release@2020.2*/
bool apq_replace(ArrayPriorityQueue* queue, void* oldKey, void* newKey, Comparator comparator, bool freeOldKey, bool freeKeyInQueue);

/*returns the size of the priority queue*/
/*asserts that queue is not NULL*/
/*since release@2020.2*/
int apq_size(ArrayPriorityQueue* queue);

/*returns the capacity of the priority queue*/
/*asserts that queue is not NULL*/
/*since release@2020.2*/
int apq_capacity(ArrayPriorityQueue* queue);

/*returns true if priority queue is empty otherwise returns false*/
/*asserts that queue is not NULL*/
/*since release@2020.2*/
bool apq_is_empty(ArrayPriorityQueue* queue);

/*returns the root of the priority queue if exist otherwise returns NULL*/
/*asserts that queue and queue's array aren't NULL*/
/*since release@2020.2*/
void* apq_root(ArrayPriorityQueue* queue);

/*returns the index of 1st occurrence of key in priority queue if exist otherwise returns -1*/
/*optionally frees the memory allocated to key at the end*/
/*asserts that queue, queue's array, callback aren't NULL*/
/*since release@2020.2*/
int apq_indexOf(ArrayPriorityQueue* queue, void* key, Comparator comparator, bool freeKeyAtEnd);

/*returns true if priority queue contains given key otherwise returns false*/
/*optionally frees the memory allocated to key at the end*/
/*asserts that queue, queue's array, callback aren't NULL*/
/*since release@2020.2*/
bool apq_contains(ArrayPriorityQueue* queue, void* key, Comparator comparator, bool freeKeyAtEnd);


/*frees the memory allocated to priority queue*/
/*optionally frees the memory allocated to keys in priority queue*/
/*asserts that queue and queue's arrays aren't NULL*/
/*since release@2020.2*/
void free_apq(ArrayPriorityQueue* queue, bool freeKeys);
