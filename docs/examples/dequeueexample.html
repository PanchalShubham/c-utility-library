utility/dqueue.h - Example

[since release@2020.1]

Implementation of de-queue (doubly ended queue) using doubly linked list!
de-queue is not syncronized!
All operations are done in constant time except for search which takes linear time!

Example

/*include necessary header file - change path according to your need*/
#include "../utility/release@2020.1/dqueue.h"

/*define how data of queue to be compared*/
/*returns -1, 0 or 1 according to key1 > key2, key1 = key2 or key1 < key2*/
int dqueueTestComparator(const void* key1, const void* key2){
    /*first we need to cast void* to their appropriate type*/
    /*we know that we have queue of integers so we cast to int* */
    int val1 = *((int*)key1);
    int val2 = *((int*)key2);

    /*make a comparison and return result*/
    return (val1 > val2 ? 1 : val1 == val2 ? 0 : -1);
}



/*print the content of queue using iterators*/
void printDequeue(DeQueue* queue){
    printf ("DeQueue Status(FRONT to REAR):\n");
    /*create an iterator for queue*/
    DeQueueIterator* iterator = new_dqueue_iterator(queue);

    /*iterate through queue (FRONT to REAR)*/
    while ( dqueue_iterator_has_next(iterator)){
        /*fetch the value from queue and point iterator to next node*/
        void* val = dqueue_iterator_next(iterator);

        /*process data - print to console*/
        /*we need to cast void* to int because it is queue of int* */
        printf ("%d ", *((int*)val));
    }
    printf ("\n");
}

/*performs tests on dequeue*/
void performDequeueTest(){

    /*create a new dequeue*/
    DeQueue* deQueue = new_dqueue();

    /*insert items in de-queue*/
    dqueue_enQueue_front(deQueue, wrap_int(30));
    dqueue_enQueue_front(deQueue, wrap_int(20));
    dqueue_enQueue_front(deQueue, wrap_int(10));
    dqueue_enQueue_rear(deQueue, wrap_int(40));
    dqueue_enQueue_rear(deQueue, wrap_int(50));
    dqueue_enQueue_rear(deQueue, wrap_int(60));

    /*print the content of de-queue to console using iterators*/
    printDequeue(deQueue);

    /*get the data from front and rear of de-queue*/
    /*note this do not delete data from de-queue*/
    void* front = dqueue_front(deQueue);
    void* rear = dqueue_rear(deQueue);
    /*so we use wrapped_int_value instead of unwrap_int because*/
    /*we just want value and we don't want to delete*/
    printf ("Front: %d and Rear: %d\n", wrapped_int_value(front), wrapped_int_value(rear));


    /*deQueue data from front of de-queue*/
    /*this will delete data from de-queue*/
    void* deQueued1 = dqueue_deQueue_front(deQueue);
    /*we use unwrap_int because data is deleted from de-queue */
    /*so we free memory  on heap*/
    printf ("deQueued: %d\n", unwrap_int(deQueued1));

    /*deQueue data from rear of de-queue*/
    /*this will delete data from de-queue*/
    void* deQueued2 = dqueue_deQueue_rear(deQueue);
    /*we use unwrap_int because data is deleted from de-queue */
    /*so we free memory  on heap*/
    printf ("deQueued: %d\n", unwrap_int(deQueued2));

    /*print content of de-queue to console*/
    printDequeue(deQueue);

    /*get index of element 10 in de-queue*/
    /*note we set freeKeyAtEnd so that memory allocated to 10 on heap is freed after search*/
    int index = dqueue_indexOf(deQueue, wrap_int(10), dqueueTestComparator, true);
    printf ("index of 10(from top of queue): %d\n", index);

    /*free memory allocated to de-queue*/
    /*also free memory allocated to all integers on queue*/
    free_dqueue(deQueue, true);

}
