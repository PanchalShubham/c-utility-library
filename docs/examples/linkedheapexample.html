utility/linkedheap.h

[since release@2020.2]

Linked binary trees implementation of binary heap!
LinkedHeap is not synchronized!
All memory allocation operations are done on heap!
Insertion and deletion takes logarithmic time!
Searching takes linear time!
since release@2020.2

Example

/*include necessary headers - modify path according to your need*/
#include "../utility/release@2020.2/linkedheap.h"
#include "../utility/release@2020.2/wrapper.h"

/*we define how our values to be compared*/
/*returns 1, 0 or -1 if object1 is greater than, equals to or smaller than object2*/
int linkedHeapTestComparator(const void* object1, const void* object2){
    /*cast objects to their appropriate type - int in this case*/
    int val1 = *((int*)object1);
    int val2 = *((int*)object2);

    /*compare values and return appropriate result*/
    return (val1 > val2 ? 1 : val1 == val2 ? 0 : -1);
}


/*prints the content in pre-order fashion for binary heap(heap)*/
void linked_heap_preOrder(LinkedHeapNode* root){
    /*check if root is NULL*/
    if (root != NULL){
        /*print the content of root - we can cast void* to int* safely*/
        /*because we know we have implemented heap of integers*/
        printf ("%d ", *((int*)(root->data)));
        /*traverse left subtree in pre-order fashion*/
        linked_heap_preOrder(root->left);
        /*traverse right subtree in pre-order fashion*/
        linked_heap_preOrder(root->right);
    }
}

/*prints the content of heap in pre-order fashion to console*/
void printLinkedHeap(LinkedHeap* heap, char* msg){
    /*print the message to user*/
    printf ("%s(size: %d)\n", msg, heap->size);
    /*print content in pre-order fashion*/
    linked_heap_preOrder(heap->root);
}

/*performs a test on linked heap*/
void performLinkedHeapTest(){

    /*create a new MAX heap with initial capacity of 10*/
    LinkedHeap* heap = new_linked_heap(LINKED_MIN_HEAP);

    /*insert 15 - elements to heap randomly*/
    printf ("Insertion order: ");
    for (int i = 0; i < 15; ++i){
        /*generate a random number for insertion in range [0...100]*/
        int data = random_int_upper_bound(100);
        linked_heap_add(heap, wrap_int(data), linkedHeapTestComparator);
        printf ("%d ", data);
    }
    /*add one determined data to heap*/
    linked_heap_add(heap, wrap_int(20), linkedHeapTestComparator);
    printf ("%d ", 20);
    printf ("\n");

    /*print the status of heap to console*/
    printLinkedHeap(heap, "Heap after insertion: "); printf ("\n");

    /*delete 13 elements from heap*/
    for (int i = 0; i < 13; ++i){
        /*extract data from heap*/
        void* data = linked_heap_extract(heap, linkedHeapTestComparator);
        /*print data to console*/
        printf ("extracted %d\n", unwrap_int(data));
    }

    /*print the status of heap to console*/
    printLinkedHeap(heap, "Heap after deletions: "); printf ("\n");

    /*check if heap contains a value 20*/
    /*note that freeKeyAtEnd is set to true so that memory allocated to 20 on heap is freed*/
    /*after the search operation is finished*/
    printf ("contains 20? %d\n", linked_heap_contains(heap, wrap_int(20), linkedHeapTestComparator, true));

    /*update the of node 20 to 100*/
    bool result = linked_heap_replace(heap, wrap_int(20), wrap_int(100), linkedHeapTestComparator, true, true);
    printf ("replaced? %s\n", result ? "true" : "false");
    /*print the status of heap to console*/
    printLinkedHeap(heap, "Heap after replacement: "); printf ("\n");


    /*print the maximum data of heap*/
    printf ("root: %d\n", wrapped_int_value(linked_heap_root(heap)));

    /*free memory allocated to heap*/
    /*also free memory allocated to keys in heap - memory allocated for ints on heap*/
    free_linked_heap(heap, true);
    printf ("Frees linkedHeap :-)\n");
}
