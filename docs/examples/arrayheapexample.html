utility/arrayheap.h

[since release@2020.2]

Resizable-array implementation of binary heaps!
ArrayHeap is not synchronized!
Capacity of the heap is doubled if array heap is 100% filled!
Capacity of the heap is halved if array heap is at most 25% filled!
Theoretical maximum capacity of array heap is INT_MAX (2,147,483,647 on most platforms)
All memory allocation operations are done on heap!
Insertion and deletion takes logarithmic time!
Searching takes linear time!
since release@2020.2

Example

/*include necessary headers*/
#include "../utility/release@2020.2/arrayheap.h"
#include "../utility/release@2020.2/wrapper.h"

/*we define how our values to be compared*/
/*returns 1, 0 or -1 if object1 is greater than, equals to or smaller than object2*/
int arrayHeapTestComparator(const void* object1, const void* object2){
    /*cast objects to their appropriate type - int in this case*/
    int val1 = *((int*)object1);
    int val2 = *((int*)object2);

    /*compare values and return appropriate result*/
    return (val1 > val2 ? 1 : val1 == val2 ? 0 : -1);
}

/*prints the content of array heap to console*/
void printArrayHeap(ArrayHeap* heap, char* msg){
    /*give a message to user*/
    printf ("%s(%d / %d)\n", msg, heap->size, heap->capacity);
    /*get the size of the heap*/
    int size = heap->size;
    /*iterate through array*/
    for (int i = 0; i < size; ++i)
        printf ("%d ", wrapped_int_value(heap->array[i]));
}

/*performs a test on array heap*/
void performArrayHeapTest(){

    /*create a new MAX heap with initial capacity of 10*/
    ArrayHeap* heap = new_array_heap(10, ARRAY_MIN_HEAP);

    /*insert 15 - elements to heap randomly*/
    /*note that capacity of heap will be expanded*/
    printf ("Insertion order: ");
    for (int i = 0; i < 15; ++i){
        /*generate a random number for insertion in range [0...100]*/
        int data = random_int_upper_bound(100);
        array_heap_add(heap, wrap_int(data), arrayHeapTestComparator);
        printf ("%d ", data);
    }
    /*add one determined data to heap*/
    array_heap_add(heap, wrap_int(20), arrayHeapTestComparator);
    printf ("%d ", 20);
    printf ("\n");

    /*print the status of heap to console*/
    printArrayHeap(heap, "Heap after insertion: "); printf ("\n");

    /*delete 13 elements from heap*/
    /*note the capacity will be shrinked*/
    for (int i = 0; i < 13; ++i){
        /*extract data from heap*/
        void* data = array_heap_extract(heap, arrayHeapTestComparator);
        /*print data to console*/
        printf ("extracted %d\n", unwrap_int(data));
    }

    /*print the status of heap to console*/
    printArrayHeap(heap, "Heap after deletions: "); printf ("\n");

    /*check if heap contains a value 20*/
    /*note that freeKeyAtEnd is set to true so that memory allocated to 20 on heap is freed*/
    /*after the search operation is finished*/
    printf ("index of 20: %d\n", array_heap_indexOf(heap, wrap_int(20), arrayHeapTestComparator, true));

    /*update the of node 20 to 100*/
    bool result = array_heap_replace(heap, wrap_int(20), wrap_int(100), arrayHeapTestComparator, true, true);
    printf ("replaced? %s\n", result ? "true" : "false");
    /*print the status of heap to console*/
    printArrayHeap(heap, "Heap after replacement: "); printf ("\n");


    /*print the maximum data of heap*/
    printf ("root: %d\n", wrapped_int_value(array_heap_root(heap)));

    /*free memory allocated to heap*/
    /*also free memory allocated to keys in heap - memory allocated for ints on heap*/
    free_array_heap(heap, true);
    printf ("Frees arrayHeap :-)\n");
}
