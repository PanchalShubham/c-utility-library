utility/timer.h

Implementation of Timer using system gettimeofday!
Designed for running time analysis of algoritms!
Timer is not syncronized!
All operations are done in constant time!

Example

/*include necessary header - change path as per your need*/
#include "../utility/release@2020.1/timer.h"
#include "../utility/release@2020.1/arrays.h"


/*define an indices comparator - how values of two arrays are to be compared*/
/*compares array[i] with key and returns -1, 0 or 1 according to less than, equals to or greater than*/
//typedef int(*ArrayIndicesComparator)(const void* array1, const void* array2, int i, int j);
int timerTestArrayIndicesComparator(const void* array1, const void* array2, int i, int j){
    /*cast array and keys to appropriate type and fetch values*/
    /*here we know we have array of integers so we cant to int* */
    int val1 = ((int*)array1)[i];
    int val2 = ((int*)array2)[j];

    /*make a comparison and return appropriate result*/
    return (val1 > val2 ? 1 : val1 == val2 ? 0 : -1);
}

/*define an indices swapper - how values at index i and j are to be swapped*/
//typedef void(*ArrayIndicesSwapper)(void* array, int i, int j);
void timerTestArrayIndicesSwapper(void* array, int i, int j){
    /*cast array and keys to appropriate type and fetch values*/
    /*here we know we have array of integers so we cant to int* */
    int val1 = ((int*)array)[i];
    int val2 = ((int*)array)[j];

    /*update values*/
    ((int*)array)[i] = val2;
    ((int*)array)[j] = val1;
}


/*performs a test on timer*/
void performTimerTest(){

    /*create a new random integer array of size 10000 on heap*/
    int N = 100000;
    int* intArray = new_random_int_array(N);

    /*create a new timer*/
    Timer* timer = new_timer();

    /*start timer before calling quick-sort*/
    timer_start(timer);
    /*sort the array[0...N - 1] using quick sort algorithm*/
    quick_sort(intArray, 0, N - 1, timerTestArrayIndicesComparator,timerTestArrayIndicesSwapper);
    /*stop timer after finishing quick-sort*/
    timer_stop(timer);

    /*get the time elapsed*/
    long long int timeElapsed = timer_time_elapsed(timer);
    printf ("Time elapsed: %lld (in us - microseconds)\n", timeElapsed);

    /*free memory allocated to array*/
    free_array(intArray);
    /*fre memory allocated to timer*/
    free_timer(timer);
}
