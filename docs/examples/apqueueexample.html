utility/apqueue.h

[since release@2020.2]

Resizable-array implementation of priority queue using binary heaps!
ArrayPriorityQueue is not synchronized!
Capacity of the queue is doubled if array priority queue is 100% filled!
Capacity of the queue is halved if array priority queue is at most 25% filled!
Theoretical maximum capacity of array priority queue is INT_MAX (2,147,483,647 on most platforms)
All memory allocation operations are done on heap!
Insertion and deletion takes logarithmic time!
Searching takes linear time!
since release@2020.2

Example

/*include necessary headers*/
#include "../utility/release@2020.2/apqueue.h"
#include "../utility/release@2020.2/wrapper.h"

/*we define how our values to be compared*/
/*returns 1, 0 or -1 if object1 is greater than, equals to or smaller than object2*/
int arrayPriorityQueueTestComparator(const void* object1, const void* object2){
    /*cast objects to their appropriate type - int in this case*/
    int val1 = *((int*)object1);
    int val2 = *((int*)object2);

    /*compare values and return appropriate result*/
    return (val1 > val2 ? 1 : val1 == val2 ? 0 : -1);
}

/*prints the content of array priority queue to console*/
void printAPQ(ArrayPriorityQueue* queue, char* msg){
    /*give a message to user*/
    printf ("%s(%d / %d)\n", msg, queue->size, queue->capacity);
    /*get the size of the queue*/
    int size = queue->size;
    /*iterate through array*/
    for (int i = 0; i < size; ++i)
        printf ("%d ", wrapped_int_value(queue->array[i]));
}

/*performs a test on array priority queue*/
void performArrayPriorityQueueTest(){
    /*create a new MAx priority queue with initial capacity of 10*/
    ArrayPriorityQueue* queue = new_apq(10, APQ_MAX);

    /*insert 15 - elements to priority queue randomly*/
    /*note that capacity of queue will be expanded*/
    printf ("Insertion order: ");
    for (int i = 0; i < 15; ++i){
        /*generate a random number for insertion in range [0...100]*/
        int data = random_int_upper_bound(100);
        apq_add(queue, wrap_int(data), arrayPriorityQueueTestComparator);
        printf ("%d ", data);
    }
    /*add one determined data to queue*/
    apq_add(queue, wrap_int(20), arrayPriorityQueueTestComparator);
    printf ("%d ", 20);
    printf ("\n");

    /*print the status of queue to console*/
    printAPQ(queue, "Queue after insertion: "); printf ("\n");

    /*delete 13 elements from priority queue*/
    /*note the capacity will be shrinked*/
    for (int i = 0; i < 13; ++i){
        /*extract data from queue*/
        void* data = apq_extract(queue, arrayPriorityQueueTestComparator);
        /*print data to console*/
        printf ("extracted %d\n", unwrap_int(data));
    }

    /*print the status of queue to console*/
    printAPQ(queue, "Queue after deletions: "); printf ("\n");

    /*check if priority queue contains a value 20*/
    /*note that freeKeyAtEnd is set to true so that memory allocated to 20 on heap is freed*/
    /*after the search operation is finished*/
    printf ("index of 20: %d\n", apq_indexOf(queue, wrap_int(20), arrayPriorityQueueTestComparator, true));

    /*update the priority of node 20 to 100*/
    bool result = apq_replace(queue, wrap_int(20), wrap_int(100), arrayPriorityQueueTestComparator, true, true);
    printf ("replaced? %s\n", result ? "true" : "false");
    /*print the status of queue to console*/
    printAPQ(queue, "Queue after replacement: "); printf ("\n");


    /*print the maximum data of priority queue*/
    printf ("root: %d\n", wrapped_int_value(apq_root(queue)));

    /*free memory allocated to queue*/
    /*also free memory allocated to keys in queue - memory allocated for ints on heap*/
    free_apq(queue, true);
    printf ("Frees arrayPriorityQueue :-)\n");
}
