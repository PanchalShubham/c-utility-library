utility/lpqueue.h

[since release@2020.2]

Linked binary heap implementation of priority queue!
LinkedPriorityQueue is not synchronized!
All memory allocation operations are done on heap!
Insertion and deletion takes logarithmic time!
Searching takes linear time!
since release@2020.2
Example

Function summary

/*creates a new linked priority queue and returns a pointer to it if succeeds otherwise returns NULL*/
/*asserts that type is valid*/
/*since release@2020.2*/
LinkedPriorityQueue* new_lpq(enum LinkedPriorityQueueType type);

/*adds given data to linked priority queue and returns true if succeeds otherwise returns false*/
/*asserts that queue and callback aren't NULL*/
/*since release@2020.2*/
bool lpq_add(LinkedPriorityQueue* pqueue, void* data, Comparator comparator);


/*returns and removes the data with max/min priority in priority queue if exist otherwise returns NULL*/
/*asserts that queue and callback aren't NULL*/
/*since release@2020.2*/
void* lpq_extract(LinkedPriorityQueue* pqueue, Comparator comparator);

/*replace the 1st occurrence of key with newKey in priority if exist; returns true if succeeds otherwise returns false*/
/*optionally frees the memory allocated to key and keyInQueue*/
/*asserts that queue, queue's array, callback aren't NULL*/
/*since release@2020.2*/
bool lpq_replace(LinkedPriorityQueue* queue, void* oldKey, void* newKey, Comparator comparator, bool freeOldKey, bool freeKeyInQueue);


/*returns the size of the linked priority queue*/
/*asserts that priority queue is not NULL*/
/*since release@2020.2*/
int lpq_size(LinkedPriorityQueue* queue);


/*returns true if priority queue is empty otherwise returns false*/
/*asserts that priority queue is not NULL*/
/*since release@2020.2*/
bool lpq_is_empty(LinkedPriorityQueue* queue);


/*returns the root of the priority queue if exist otherwise returns NULL*/
/*asserts that queue isn't NULL*/
/*since release@2020.2*/
void* lpq_root(LinkedPriorityQueue* queue);

/*returns true if priority queue contains given key otherwise returns false*/
/*optionally frees the memory allocated to key at the end*/
/*asserts that queue and callback aren't NULL*/
/*since release@2020.2*/
bool lpq_contains(LinkedPriorityQueue* queue, void* key, Comparator comparator, bool freeKeyAtEnd);


/*frees the memory allocated to priority queue*/
/*optionally frees the memory  allocated to nodes*/
/*asserts that queue is not empty*/
void free_lpq(LinkedPriorityQueue* queue, bool freeKeys);
