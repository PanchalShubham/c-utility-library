utility/random.h - Example

Wrapper for random number/string generators!

Example

/*include necessary header - change path as per your need*/
#include "../utility/release@2020.1/random.h"

/*performs a test for random header*/
void performRandomTest(){

    /*seed random number generator*/
    randomize();

    /*print a random integer to console*/
    printf ("%d\n", random_int());
    /*print a random integer to console in range [0...100]*/
    printf ("%d\n", random_int_upper_bound(100));
    /*print a random integer to console in range [10...50]*/
    printf ("%d\n", random_int_range(10, 50));

    /*print a random double in range [0...1]*/
    printf ("%lf\n", random_double());
    /*print a random double in range [0...1.5]*/
    printf ("%lf\n", random_double_upper_bound(1.5));
    /*print a random double in range [1.5...2.5]*/
    printf ("%lf\n", random_double_range(1.5, 2.5));



    /*stores temporarily strings*/
    char* string = NULL;
    /*generate a random binary string and print to console*/
    string = random_binary_string(10);
    printf ("%s\n", string);
    /*free memory allocated to binary string on heap*/
    free (string);


    /*generate a random octal string and print to console*/
    string = random_octal_string(10);
    printf ("%s\n", string);
    /*free memory allocated to octal string on heap*/
    free (string);

    /*generate a random decimal and print to console*/
    string = random_decimal_string(10);
    printf ("%s\n", string);
    /*free memory allocated to decimal string on heap*/
    free (string);

    /*generate a random hexadecimal string and print to console*/
    string = random_hex_string(10);
    printf ("%s\n", string);
    /*free memory allocated to hexadecimal string on heap*/
    free (string);

    /*generate a random printable string and print to console*/
    string = random_printable_string(10);
    printf ("%s\n", string);
    /*free memory allocated to printable string on heap*/
    free (string);

    /*generate a random alphabetical string and print to console*/
    string = random_alphabetical_string(10);
    printf ("%s\n", string);
    /*free memory allocated to alphabetical string on heap*/
    free (string);

    /*generate a random alphanumeric string and print to console*/
    string = random_alphanumeric_string(10);
    printf ("%s\n", string);
    /*free memory allocated to alphanumeric string on heap*/
    free (string);

    /*generate a random special character string and print to console*/
    string = random_special_string(10);
    printf ("%s\n", string);
    /*free memory allocated to special character string on heap*/
    free (string);

}
